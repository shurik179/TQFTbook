% creates a "cloud" of coordinates around a given coordinate, adding suffixes "t", "b","l", "r"
\newcommand{\makecoords}[1]{%
    \path (#1) -- +(100:0.3 and 0.1) coordinate (#1t);
    \path (#1) -- +(-80:0.3 and 0.1) coordinate (#1b);
    \path (#1) -- +(0.3,0) coordinate (#1r);
    \path (#1) -- +(-0.3,0) coordinate (#1l);
}
%% creates partly dashed path.
%% argument 1: softpath
%% argument 2: clipping path (e.g. (0,0) rectanlge (1,1)); everything inside that
%% path will be dashed

\newcommand{\partlydashed}[2]{
  \begin{scope}[on background layer]
    \draw[spath/use=#1];
  \end{scope}
  \begin{scope}
    \clip #2;
    \draw[white, line width =0.42pt, spath/use=#1];
    \draw[cobdash, spath/use=#1];
  \end{scope}
}


\tikzset{%
  ellipse top/.style = {insert path ={+(0:0.3) arc (0:180:0.3 and 0.1) }},
  ellipse bottom/.style = {insert path ={+(180:0.3) arc (180:360:0.3 and 0.1)} },
  ellipse full/.style = {insert path ={circle (0.3 and 0.1)} },
  elbow right/.style = {insert path ={ arc (100:-80:0.3 and 0.1) -- +(-0.0,0)}},
  elbow left/.style = {insert path ={ arc (100:280:0.3 and 0.1) -- +(0.0,0) }},
}

%%%% boolean variables, used for enabling/disabling
%%%% of drawing of top/bottom ellipses in cobordisms
\newif\ifnotop
\newif\ifnobottom
\tikzset{
  notop/.is if = notop,
  notop = false,
  nobottom/.is if = nobottom,
  nobottom = false,
}

%%% custom length units
\newlength{\cobheight}    % Height of a pair of pants
\setlength{\cobheight}{1.2cm}
\tikzset{
   cobheight/.code  = {\setlength{\cobheight}{#1}},
   cobheight/.default = 1.2cm,
}

\newlength{\cobshift}    % Shift of a bottom of cylinder
\setlength{\cobshift}{0cm}
\tikzset{
  cobshift/.code  = {\setlength{\cobshift}{#1}},
  cobshift/.default = 0cm,
}

%%%% custom syles, empty by default
\tikzset {
  cobtop/.style = {},        %for top holes
  cobbottom/.style = {}, % for bottom holes
  cobside/.style = {},      % for sides
}

%%%% Shapes.
%%%% pants (1 to 2), triplepants (1 to 3), copants (2 to 1), cup (1 to 0), cap
%%%%(0 to 1), pairing (2 to 0), copairing (0 to 2), cylinder
\tikzset {
  pants/.pic ={
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (-0.5,-\cobheight);
    \coordinate (-bottom2) at (0.5,-\cobheight);
    \coordinate (-center) at (0, -0.5\cobheight);
    % draw top ellipse, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
    \fi
    % draw bottom ellipses, unless nobottom option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom] (-bottom2) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
      \draw[cobbottom,cobdash] (-bottom2) [ellipse top];
    \fi
    \draw (-0.8, -\cobheight) to [out=90, in=270, looseness=1] (-0.3,0);
    \draw (0.8,-\cobheight) to [out=90, in=270, looseness=1] (0.3,0);
    \coordinate (b2) at (0.2,-\cobheight);
    \draw (-0.2, -\cobheight) to [out=90, in=90,  looseness=0.09\cobheight] (b2);
  },
  triplepants/.pic ={
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (-1,-\cobheight);
    \coordinate (-bottom2) at (0,-\cobheight);
    \coordinate (-bottom3) at (1,-\cobheight);
    \coordinate (-center) at (0, -0.5\cobheight);
    % draw top ellipse, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
    \fi
    % draw bottom ellipses, unless nobottom option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom] (-bottom2) [ellipse bottom];
      \draw[cobbottom] (-bottom3) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
      \draw[cobbottom,cobdash] (-bottom2) [ellipse top];
      \draw[cobbottom,cobdash] (-bottom3) [ellipse top];
    \fi
    \draw (-1.3, -\cobheight) to [out=90, in=270, looseness=1] (-0.3,0);
    \draw (1.3,-\cobheight) to [out=90, in=270, looseness=1] (0.3,0);
    \coordinate (b2) at (-0.3,-\cobheight);
    \draw (-0.7, -\cobheight) to [out=90, in=90,  looseness=0.09\cobheight]
    (b2);
    \coordinate (b3) at (0.3,-\cobheight);
    \draw (0.7, -\cobheight) to [out=90, in=90,  looseness=0.09\cobheight]
    (b3);
  },
  copants/.pic ={
    \coordinate (-top1) at (0,0);
    \coordinate (-top2) at (1,0);
    \coordinate (-bottom1) at (0.5,-\cobheight);
    \coordinate (-center) at (0.5, -0.5\cobheight);
    % draw top ellipses, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
      \draw[cobtop] (-top2) [ellipse full];
    \fi
    % draw bottom ellipse, unless nobottom option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
    \fi
    \draw (0.2, -\cobheight) to [out=90, in=270, looseness=1] (-0.3,0);
    \draw (0.8,-\cobheight) to [out=90, in=270, looseness=1] (1.3,0);
    \draw (0.3,0) to [out=270, in=270,  looseness=0.09\cobheight] (0.7,0);
  },
  pairing/.pic ={
    \coordinate (-top1) at (0,0);
    \coordinate (-top2) at (1,0);
    \coordinate (-center) at (0.5, -0.4\cobheight);
    % draw top ellipses, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
      \draw[cobtop] (-top2) [ellipse full];
    \fi
    \draw (0.3,0) to [out=270, in=270,  looseness=0.07\cobheight] (0.7,0);
    \draw (-0.3,0) to [out=270, in=270,  looseness=0.055\cobheight] (1.3,0);
  },
  copairing/.pic ={
    \coordinate (-bottom1) at (0,0);
    \coordinate (-bottom2) at (1,0);
    \coordinate (-center) at (0.5, 0.4\cobheight);
    % draw top ellipses, unless notop option was given
    % draw bottom ellipses, unless nobottom option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom] (-bottom2) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
      \draw[cobbottom,cobdash] (-bottom2) [ellipse top];
    \fi
    \draw (0.3,0) to [out=90, in=90,  looseness=0.07\cobheight] (0.7,0);
    \draw (-0.3,0) to [out=90, in=90,  looseness=0.055\cobheight] (1.3,0);
  },
  cup/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-center) at (0, -0.12\cobheight);
    % draw top ellipse, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
    \fi
    \draw(-0.3,0) to [out = 270, in =270, looseness = 0.06\cobheight] (0.3,0);
  },
  cap/.pic = {
    \coordinate (-bottom1) at (0,0);
    \coordinate (-center) at (0, 0.12\cobheight);
    % draw bottom ellipse, unless nobottom option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
    \fi
    \draw(-0.3,0) to [out = 90, in=90, looseness = 0.06\cobheight] (0.3,0);
  },
  cylinder/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (\cobshift,-\cobheight);
    \coordinate (-center) at (0.5\cobshift, -0.5\cobheight);
    \coordinate (b1) at({\cobshift-0.3cm},-\cobheight);
    \coordinate (b2) at({\cobshift+0.3cm},-\cobheight);
    % draw top ellipse, unless notop option was given
    \ifnotop
    \else
      \draw[cobtop] (-top1) [ellipse full];
    \fi
    % draw bottom ellipse, unless notop option was given
    \ifnobottom
    \else
      \draw[cobbottom] (-bottom1) [ellipse bottom];
      \draw[cobbottom,cobdash] (-bottom1) [ellipse top];
    \fi
    \draw (b1) to [out=90, in=270, looseness=0.9] (-0.3, 0);
    \draw (b2) to [out=90, in=270, looseness=0.9] (0.3, 0);
  },
  halfcylinder_r/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (0,-\cobheight);
    %now add clouds around these points
    \makecoords{-top1};
    \makecoords{-bottom1};
    %top
    \ifnotop
    \else
      \draw[cobtop] (-top1t) [elbow right];
    \fi
    %bottom
    \path[spath/save=bot] (-bottom1t) [elbow right];
    \partlydashed{bot}{(-top1b) rectangle (-bottom1r)}
    %left
    \draw (-top1t)--(-bottom1t);
    \draw (-top1b)--(-bottom1b);
    %right
    \draw (-top1r)--(-bottom1r);
  },
  halfcylinder_l/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (0,-\cobheight);
    %now add clouds around these points
    \makecoords{-top1}
    \makecoords{-bottom1}
    %top & right
    \path[spath/save=right] (-top1t)--(-bottom1t);
    \ifnotop
      %%draw right side dasehd, and do not draw top
      \draw[cobdash, spath/use = right];
    \else
      %% draw top, and draw right partly dashed
      \draw[cobtop] (-top1t) [elbow left];
      \partlydashed{right}{(-top1b) rectangle (-bottom1l)}
    \fi
    %bottom
    \path[spath/save=bot] (-bottom1t) [elbow left];
    \partlydashed{bot}{(-top1b) rectangle (-bottom1l)}
    %left
    \draw (-top1l)--(-bottom1l);
    %right
    \draw (-top1b)--(-bottom1b);
  },
  saddle/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-top2) at (1,0);
    \coordinate (-bot1) at (0,-\cobheight);
    \coordinate (-bot2) at (1,-\cobheight);
    %now add clouds around these points
    \makecoords{-top1}
    \makecoords{-top2}
    \makecoords{-bot1}
    \makecoords{-bot2}
    %% rear left and  bottom: create path, then make it dashed
    \path[spath/save=backface] (-top1t)--(-bot1t)--(-bot2t);
    \partlydashed{backface}{(-bot1b) rectangle (-top2b)}
    %% top and rear right
    \ifnotop
      \draw[cobdash] (-bot2t)--(-top2t);
    \else
      \path[spath/save=backright] (-top2t)--(-bot2t);
      \partlydashed{backright}{(-bot1b) rectangle (-top2b)}
      \draw[cobtop] (-top1t) [elbow right];
      \draw[cobtop] (-top2t) [elbow left];
    \fi
    \draw (0.3,0) to [out=270, in=270,  looseness=0.12\cobheight] (0.7,0);
    %% front face
    \draw (-top1b)--(-bot1b)--(-bot2b)--(-top2b);

  },
  revsaddle/.pic = {
    \coordinate (-top1) at (0,0);
    \coordinate (-top2) at (1,0);
    \coordinate (-bot1) at (0,-\cobheight);
    \coordinate (-bot2) at (1,-\cobheight);
    %now add clouds around these points
    \makecoords{-top1}
    \makecoords{-top2}
    \makecoords{-bot1}
    \makecoords{-bot2}
    \ifnotop
    \else
      \draw[cobtop] (-top1t) -- (-top2t);
      \draw[cobtop] (-top1b) -- (-top2b);
    \fi
    %% draw back face
    %% rear face: create path, then make it dashed
    \path[spath/save=backface] (-bot1t) [elbow right] (-bot2t) [elbow left] (-top1t)--(-bot1t) (-top2t)--(-bot2t);
    \partlydashed{backface}{(-top1b) rectangle (-bot2)}
    %% curve
    \coordinate (bot2left) at (0.7,-\cobheight);
    \draw (0.3,-\cobheight) to [out = 90, in = 90, looseness=0.12\cobheight] (bot2left);
    %% front face
    \draw (-top1b) -- (-bot1b) (-top2b)--(-bot2b);
  },
  zigzag1/.pic = {
    \draw (0,0)--(0.5,0) [elbow right] coordinate (-mid) [elbow left] -- +(0.5,0) coordinate (-top2);
    \coordinate (-rightelbow) at ($(-mid) + (0.25,0.1)$);
    \coordinate (-leftelbow) at ($(-mid) + (-0.25,-0.1)$);
  },
  cusp1/.pic = {
    %top
    \coordinate (-top1) at (0,0);
    \makecoords{-top1}
    \path[spath/save=zigzag]  (0,0)--(0.5,0) [elbow right] coordinate (-mid) [elbow left] -- +(0.5,0) coordinate (-top2);
    \coordinate (-sing) at ($(-mid) + (0,-0.7)$);
    %% right crease line
    \path[spath/save=crease2]  (-sing) to [out = 90, in = -90] +(0.25,0.8);
    % sides and bottom
    \draw (-top2) -- +(0, -\cobheight) coordinate (-bottom2);
    \coordinate (-bottom1) at (-top1 |- -bottom2);
    \draw (-top1) -- (-bottom1);
    \draw (-bottom1)--(-bottom2);
    %top
    \ifnotop
      \draw[cobdash, spath/use = crease2];
    \else
      \draw[cobtop, spath/use = zigzag];
       %%one crease line is partly dashed
      \partlydashed{crease2}{(-bottom1) rectangle (-top2)}
    \fi
    %% now the crease lines
    \draw (-sing) to [out = 90, in = -90]  +(-0.25,0.6);
  },
  cusp2/.pic = {
    %bottom
    \coordinate (-top1) at (0,0);
    \coordinate (-bottom1) at (0,-\cobheight);
    \path[spath/save=zigzag] (-bottom1)--+(0.5,0) [elbow right] coordinate (-mid) [elbow left] -- +(0.5,0) coordinate (-bottom2);
    \coordinate (-top2) at (-top1 -| -bottom2);
    \coordinate (-sing) at ($(-mid) + (0,0.7)$);
    \coordinate (-creaseleft) at ($(-sing) + (-0.25,-0.8)$);
    \ifnotop
    \else
      \draw[cobtop] (-top1) -- (-top2);
    \fi
    % draw bottom zigag, partly dashed
    \partlydashed{zigzag}{(-sing)-- (-creaseleft) -- +(2,0) -- (-sing)}
    %sides
    \draw (-top1) -- (-bottom1);
    \draw (-top2)--(-bottom2);
    %\node[dotnode] at (-sing) {};
    \draw (-sing) to [out = -90, in = 90]  (-creaseleft);
    \draw[cobdash] (-sing) to [out = -90, in = 90]  +(0.25,-0.6);
  },
}
